from tkinter import *  #Импорт файла

canvas_width = 700  #Размер окна
canvas_height = 500
brush_size = 3 #Размер кисти
brish_size_change = 5
color = "black" #Цвет кисти


def paint(event): #Функция рисования
    global brush_size
    global color
    x1 = event.x - brush_size # По X
    x2 = event.x + brush_size
    y1 = event.y - brush_size # По Y
    y2 = event.y + brush_size
    w.create_oval(x1, y1, x2, y2,
                  fill=color, outline=color)
    
    def brush_size_change(new_size):
        global brush_size
        brush_size = new_size

def color_change (new_color):
    global color
    color = new_color
    
root = Tk()
root.title("paint на Питоне") # Название окна
w = Canvas(root,
           width=canvas_width,
           height=canvas_height,
           bg="white")
w.bind("<B1-Motion>", paint) # Основная функция
##############КНОПКИ####################

purple_btn = Button(text='Фиолетовый', width=10,
                 command =lambda:
                 color_change('purple'))

blue_btn = Button(text='Синий', width=10,
                 command =lambda:
                 color_change('blue'))

white_btn = Button(text='Ластик', width=10,
                 command =lambda:
                 color_change('white'))

clear_btn = Button (text ='Стереть все', width=10,
                    command=lambda: w.delete('all'))

five_btn = Button (text = 'Размер 5', width=10,
                  command=lambda: brish_size_change(5))

w.grid(row=2, column=0,          #Хз что
       columnspan=7, padx=5,
       pady=5, sticky=E+W+S+N)
w.columnconfigure(6, weight=1)
w.rowconfigure(2, weight=1)

purple_btn.grid(row=0, column=2)
blue_btn.grid(row=0, column=3)
white_btn.grid(row=0, column=4)
clear_btn.grid(row=0, column=5, sticky=W)
five_btn.grid(row=1, column=6)

root.mainloop()
